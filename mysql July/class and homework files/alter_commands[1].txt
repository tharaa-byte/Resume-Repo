-- Create the initial table
CREATE TABLE employeees (
    employee_id INT PRIMARY KEY,
    first_name VARCHAR(50),
    last_name VARCHAR(50),
    salary DECIMAL(10, 2)
);

-- Add a new column 'department'
ALTER TABLE employeees ADD COLUMN department VARCHAR(50);

-- Modify the 'salary' column to increase its size and add a NOT NULL constraint
ALTER TABLE employees MODIFY COLUMN salary DECIMAL(15, 2) NOT NULL;

-- Drop the 'department' column
ALTER TABLE employees DROP COLUMN department;

-- Add a new column 'email' and add a UNIQUE constraint to it
ALTER TABLE employeees ADD COLUMN email VARCHAR(100);
ALTER TABLE employees ADD CONSTRAINT unique_email UNIQUE (email);

--Dropping a PRIMARY KEY
ALTER TABLE employeees DROP PRIMARY KEY;
Many DBMSs have designed the DROP PRIMARY KEY command to simplify the process of removing primary key constraints. 
Since a table cannot have more than one primary key, 
the command knows which constraint to drop without needing additional information.


-- Rename the table 'employees' to 'staff'
ALTER TABLE employees RENAME TO staff;

-- Rename the 'first_name' column to 'fname'
ALTER TABLE employees RENAME COLUMN first_name TO fname;

1. Dropping a PRIMARY KEY
ALTER TABLE table_name DROP PRIMARY KEY;
2. Dropping a FOREIGN KEY
First, find the name of the foreign key constraint (e.g., using SHOW CREATE TABLE), then drop it.
ALTER TABLE table_name DROP FOREIGN KEY constraint_name;
3. Dropping a UNIQUE Constraint
The most common way is to drop the index associated with the unique constraint.
ALTER TABLE table_name DROP INDEX index_name;
4. Dropping a CHECK Constraint
Starting from MySQL 8.0.16, you can use:
ALTER TABLE table_name DROP CONSTRAINT constraint_name;


CREATE TABLE example_table (
    id INT PRIMARY KEY,
    username VARCHAR(50) UNIQUE,
    email VARCHAR(100) UNIQUE,
    age INT,
    department_id INT,
    CONSTRAINT fk_department FOREIGN KEY (department_id) REFERENCES departmentsss(department_id),
    CONSTRAINT check_age CHECK (age >= 18)
);

CREATE TABLE departmentsss (
    department_id INT PRIMARY KEY,
    department_name VARCHAR(50) NOT NULL
);

Dropping the PRIMARY KEY
ALTER TABLE example_table DROP PRIMARY KEY;

Dropping the FOREIGN KEY
ALTER TABLE example_table DROP FOREIGN KEY fk_department;

Dropping the UNIQUE Constraint
ALTER TABLE example_table DROP INDEX username;
ALTER TABLE example_table DROP INDEX email;

Dropping the CHECK Constraint
From MySQL 8.0.16 onward:
ALTER TABLE example_table DROP CONSTRAINT check_age;
