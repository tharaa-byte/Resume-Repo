

_________________________________________STRING FUNCTION_________________________________________

1. ASCII
Returns the ASCII value for the specific character.

-> SELECT ASCII('a') AS ascii_value;  -- Output: 97

2. CHAR_LENGTH
Returns the length of a string (in characters).
  
-> SELECT CHAR_LENGTH('Hello') AS char_length;  -- Output: 5


3. CHARACTER_LENGTH
Returns the length of a string (in characters).

-> SELECT CHARACTER_LENGTH('World') AS char_length;  -- Output: 5


4. CONCAT
Adds two or more expressions together.

-> SELECT CONCAT('Hello', ' ', 'World') AS concatenated_string;  -- Output: 'Hello World'


5. FORMAT
Formats a number to a format like "#,###,###.##", rounded to a specified number of decimal places.

-> SELECT FORMAT(12345.678, 2) AS formatted_number;  -- Output: '12,345.68'


6. INSERT
Inserts a string within a string at the specified position and for a certain number of characters.

-> SELECT INSERT('Hello World', 7, 5, 'Everyone') AS inserted_string;  -- Output: 'Hello Everyone'


7. LCASE
Converts a string to lower-case.

-> SELECT LCASE('HELLO WORLD') AS lower_case_string;  -- Output: 'hello world'


8. LEFT
Extracts a number of characters from a string (starting from left).

-> SELECT LEFT('Hello World', 5) AS left_string;  -- Output: 'Hello'


9. LENGTH
Returns the length of a string (in bytes).

-> SELECT LENGTH('Hello') AS length_bytes;  -- Output: 5


10. LOCATE
Returns the position of the first occurrence of a substring in a string.

-> SELECT LOCATE('World', 'Hello World') AS locate_position;  -- Output: 7

11. LOWER
Converts a string to lower-case.

-> SELECT LOWER('HELLO WORLD') AS lower_case_string;  -- Output: 'hello world'


12. LPAD
Left-pads a string with another string, to a certain length.

-> SELECT LPAD('Hello', 10, ' ') AS lpad_string;  -- Output: '     Hello'


13. LTRIM
Removes leading spaces from a string.

-> SELECT LTRIM('   Hello') AS ltrim_string;  -- Output: 'Hello'


14. MID
Extracts a substring from a string (starting at any position).

-> SELECT MID('Hello World', 7, 5) AS mid_string;  -- Output: 'World'

15. POSITION
Returns the position of the first occurrence of a substring in a string.

-> SELECT POSITION('World' IN 'Hello World') AS position;  -- Output: 7


16. REPEAT
Repeats a string as many times as specified.

-> SELECT REPEAT('Hello', 3) AS repeated_string;  -- Output: 'HelloHelloHello'

17. REPLACE
Replaces all occurrences of a substring within a string, with a new substring.

-> SELECT REPLACE('Hello World', 'World', 'Everyone') AS replaced_string;  -- Output: 'Hello Everyone'

18. REVERSE
Reverses a string and returns the result.

-> SELECT REVERSE('Hello') AS reversed_string;  -- Output: 'olleH'

19. RIGHT
Extracts a number of characters from a string (starting from right).

-> SELECT RIGHT('Hello World', 5) AS right_string;  -- Output: 'World'

20. RTRIM
Removes trailing spaces from a string.

-> SELECT RTRIM('Hello   ') AS rtrim_string;  -- Output: 'Hello'

21. SPACE
Returns a string of the specified number of space characters.

-> SELECT SPACE(5) AS space_string;  -- Output: '     ' (5 spaces)

22. SUBSTR
Extracts a substring from a string (starting at any position).

-> SELECT SUBSTR('Hello World', 7, 5) AS substr_string;  -- Output: 'World'

23. SUBSTRING
Extracts a substring from a string (starting at any position).

-> SELECT SUBSTRING('Hello World', 7, 5) AS substring_string;  -- Output: 'World'

24. TRIM
Removes leading and trailing spaces from a string.

-> SELECT TRIM('   Hello World   ') AS trim_string;  -- Output: 'Hello World'

25. UCASE
Converts a string to upper-case.

-> SELECT UCASE('hello world') AS upper_case_string;  -- Output: 'HELLO WORLD'


26. UPPER
Converts a string to upper-case

-> SELECT UPPER('hello world') AS upper_case_string;  -- Output: 'HELLO WORLD'


____________________________________numerical functions________________________________________________________

ABS
Description: Returns the absolute value of a number.

SELECT ABS(-5) AS abs_value;  -- Output: 5 abs(Non-Ngative)

AVG
Description: Returns the average value of an expression.

SELECT AVG(salary) AS avg_salary FROM employees;


CEIL
Description: Returns the smallest integer value that is >= to a number.

SELECT CEIL(4.2) AS ceil_value;  -- Output: 5 (if ceil(4.0) it will show 4 as output)

CEILING
Description: Returns the smallest integer value that is >= to a number.

SELECT CEILING(4.2) AS ceiling_value;  -- Output: 5


COUNT
Description: Returns the number of records returned by a select query.

SELECT COUNT(*) AS total_records FROM employees;


DIV
Description: Used for integer division.

SELECT 9 DIV 4 AS div_result;  -- Output: 2


FLOOR
Description: Returns the largest integer value that is <= to a number.

SELECT FLOOR(4.7) AS floor_value;  -- Output: 4



MAX
Description: Returns the maximum value in a set of values.

SELECT MAX(salary) AS max_salary FROM employees;


MIN
Description: Returns the minimum value in a set of values.

SELECT MIN(salary) AS min_salary FROM employees;


MOD
Description: Returns the remainder of a number divided by another number.

SELECT MOD(10, 3) AS mod_result;  -- Output: 1

PI
Description: Returns the value of PI.

SELECT PI() AS pi_value;  -- Output: 3.141592653589793

POW
Description: Returns the value of a number raised to the power of another number.

SELECT POW(2, 3) AS pow_result;  -- Output: 8

POWER
Description: Returns the value of a number raised to the power of another number.

SELECT POWER(2, 3) AS power_result;  -- Output: 8


ROUND
Description: Rounds a number to a specified number of decimal places.

SELECT ROUND(4.567, 2) AS round_value;  -- Output: 4.57

SIGN
Description: Returns the sign of a number.

SELECT SIGN(-5) AS sign_value;  -- Output: -1
SELECT SIGN(0) AS sign_value;   -- Output: 0
SELECT SIGN(5) AS sign_value;   -- Output: 1

SQRT
Description: Returns the square root of a number.

SELECT SQRT(16) AS sqrt_value;  -- Output: 4

SUM
Description: Calculates the sum of a set of values.

SELECT SUM(salary) AS total_salary FROM employees;


_________________________________________ Date and time functions _____________________________________________________________________




Function	Description


CURDATE	Returns the current date
-> SELECT CURDATE() AS current_date;  -- Output: Current Date (e.g., 2024-05-30)


CURRENT_DATE	Returns the current date
-> SELECT CURRENT_DATE() AS current_date;  -- Output: Current Date (e.g., 2024-05-30)


CURRENT_TIME	Returns the current time
-> SELECT CURRENT_TIME() AS current_time;  -- Output: Current Time (e.g., 14:35:20)


CURRENT_TIMESTAMP	Returns the current date and time
-> SELECT CURRENT_TIMESTAMP() AS current_datetime;  -- Output: Current DateTime (e.g., 2024-05-30 14:35:20)


CURTIME	Returns the current time
-> SELECT CURTIME() AS current_time;  -- Output: Current Time (e.g., 14:35:20)

DATE	Extracts the date part from a datetime expression
-> SELECT DATE('2024-05-30 14:35:20') AS date_part;  -- Output: 2024-05-30

DATEDIFF	Returns the number of days between two date values
-> SELECT DATEDIFF('2024-05-30', '2024-05-25') AS day_difference;  -- Output: 5

DATE_FORMAT	Formats a date
-> SELECT DATE_FORMAT('2024-05-30', '%W, %M %d, %Y') AS formatted_date;  -- Output: Thursday, May 30, 2024

DAY	Returns the day of the month for a given date
-> SELECT DAY('2024-05-30') AS day_of_month;  -- Output: 30

DAYNAME	Returns the weekday name for a given date
-> SELECT DAYNAME('2024-05-30') AS weekday_name;  -- Output: Thursday

DAYOFMONTH	Returns the day of the month for a given date
-> SELECT DAYOFMONTH('2024-05-30') AS day_of_month;  -- Output: 30

DAYOFWEEK	Returns the weekday index for a given date
-> SELECT DAYOFWEEK('2024-05-30') AS weekday_index;  -- Output: 5 (Sunday=1, Saturday=7)

DAYOFYEAR	Returns the day of the year for a given date
-> SELECT DAYOFYEAR('2024-05-30') AS day_of_year;  -- Output: 151

EXTRACT	Extracts a part from a given date
-> SELECT EXTRACT(YEAR FROM '2024-05-30') AS year_part;  -- Output: 2024
-> SELECT EXTRACT(MONTH FROM '2024-05-30') AS month_part;  -- Output: 5
-> SELECT EXTRACT(DAY FROM '2024-05-30') AS day_part;  -- Output: 30

FROM_DAYS	Returns a date from a numeric datevalue
-> SELECT FROM_DAYS(738100) AS date_value;  -- Output: 2024-05-30

HOUR	Returns the hour part for a given date
-> SELECT HOUR('2024-05-30 14:35:20') AS hour_part;  -- Output: 14

LAST_DAY	Extracts the last day of the month for a given date
-> SELECT LAST_DAY('2024-05-30') AS last_day_of_month;  -- Output: 2024-05-31

LOCALTIME	Returns the current date and time
-> SELECT LOCALTIME() AS local_datetime;  -- Output: Current DateTime (e.g., 2024-05-30 14:35:20)

LOCALTIMESTAMP	Returns the current date and time
-> SELECT LOCALTIMESTAMP() AS local_timestamp;  -- Output: Current DateTime (e.g., 2024-05-30 14:35:20)

MAKEDATE	Creates and returns a date based on a year and a number of days value
-> SELECT LOCALTIMESTAMP() AS local_timestamp;  -- Output: Current DateTime (e.g., 2024-05-30 14:35:20)

MAKETIME	Creates and returns a time based on an hour, minute, and second value
-> SELECT MAKETIME(14, 35, 20) AS make_time;  -- Output: 14:35:20

MICROSECOND	Returns the microsecond part of a time/datetime
-> SELECT MICROSECOND('2024-05-30 14:35:20.123456') AS microsecond_part;  -- Output: 123456

MINUTE	Returns the minute part of a time/datetime
-> SELECT MINUTE('2024-05-30 14:35:20') AS minute_part;  -- Output: 35

MONTH	Returns the month part for a given date
-> SELECT MONTH('2024-05-30') AS month_part;  -- Output: 5

MONTHNAME	Returns the name of the month for a given date
-> SELECT MONTHNAME('2024-05-30') AS month_name;  -- Output: May

NOW	Returns the current date and time
-> SELECT NOW() AS current_datetime;  -- Output: Current DateTime (e.g., 2024-05-30 14:35:20)

PERIOD_ADD	Adds a specified number of months to a period
-> SELECT PERIOD_ADD(202405, 1) AS new_period;  -- Output: 202406 (2024 May + 1 month)

PERIOD_DIFF	Returns the difference between two periods
-> SELECT PERIOD_DIFF(202405, 202304) AS period_difference;  -- Output: 13 (months between April 2023 and May 2024)

QUARTER	Returns the quarter of the year for a given date value
-> SELECT QUARTER('2024-05-30') AS quarter_part;  -- Output: 2

SECOND	Returns the seconds part of a time/datetime
-> SELECT SECOND('2024-05-30 14:35:20') AS second_part;  -- Output: 20

SEC_TO_TIME	Returns a time value based on the specified seconds
-> SELECT SEC_TO_TIME(3665) AS time_value;  -- Output: 01:01:05

STR_TO_DATE	Returns a date based on a string and a format
-> SELECT STR_TO_DATE('30-May-2024', '%d-%b-%Y') AS date_value;  -- Output: 2024-05-30

SUBTIME	Subtracts a time interval from a datetime and then returns the time/datetime
-> SELECT SUBTIME('14:35:20', '01:30:00') AS new_time;  -- Output: 13:05:20

SYSDATE	Returns the current date and time
-> SELECT SYSDATE() AS system_datetime;  -- Output: Current DateTime (e.g., 2024-05-30 14:35:20)

TIME	Extracts the time part from a given time/datetime
-> SELECT TIME('2024-05-30 14:35:20') AS time_part;  -- Output: 14:35:20

TIME_FORMAT	Formats a time by a specified format
-> SELECT TIME_FORMAT('14:35:20', '%h:%i %p') AS formatted_time;  -- Output: 02:35 PM

TIME_TO_SEC	Converts a time value into seconds
-> SELECT TIME_TO_SEC('01:01:05') AS seconds_value;  -- Output: 3665

TIMEDIFF	Returns the difference between two time/datetime expressions
-> SELECT TIMEDIFF('2024-05-30 14:35:20', '2024-05-30 13:30:00') AS time_difference;  -- Output: 01:05:20

TO_DAYS	Returns the number of days between a date and date "0000-00-00"
-> SELECT TO_DAYS('2024-05-30') AS days_value;  -- Output: 738100

WEEK	Returns the week number for a given date
-> SELECT WEEK('2024-05-30') AS week_number;  -- Output: 22

WEEKDAY	Returns the weekday number for a given date
-> SELECT WEEKDAY('2024-05-30') AS weekday_number;  -- Output: 3 (Monday=0, Sunday=6)

WEEKOFYEAR	Returns the week number for a given date
-> SELECT WEEKOFYEAR('2024-05-30') AS week_of_year;  -- Output: 22

YEAR	Returns the year part for a given date
-> SELECT WEEKOFYEAR('2024-05-30') AS week_of_year;  -- Output: 22

YEARWEEK	Returns the year and week number for a given date
-> SELECT YEARWEEK('2024-05-30') AS year_week;  -- Output: 202422
